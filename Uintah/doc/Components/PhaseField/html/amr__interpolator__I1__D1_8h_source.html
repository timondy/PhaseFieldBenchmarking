140  Region
141  compute_fine_region (
142  const IntVector & low,
143  const IntVector & high
144  )
145  {
146  IntVector r = m_level_fine->getRefinementRatio();
147  IntVector l_ghosts {0, 0, 0}, h_ghosts {0, 0, 0};
148  for ( size_t d = 0; d < DIM; ++d )
149  {
150  int total = r[d] * ( E - 1 );
151  h_ghosts[d] = total / 2;
152  l_ghosts[d] = total - h_ghosts[d];
153  if ( Problem::VAR == NC )
154  {
155  if ( !low[d] % r[d] && l_ghosts[d] > 1 ) l_ghosts[d] -= 1;
156  if ( !high[d] % r[d] && h_ghosts[d] > 1 ) h_ghosts[d] -= 1;
157  }
158  }
159 
160  Region fine_region;
161  fine_region.low() = low - l_ghosts;
162  fine_region.high() = high + h_ghosts;
163 
164  return fine_region;
165  }

400  virtual T
401  operator[] (
402  const IntVector & id_fine
403  ) const override
404  {
405  IntVector id_coarse ( m_level_fine->mapCellToCoarser ( id_fine ) );
406  Point p_fine ( DWInterface<Problem::VAR, Problem::DIM>::get_position ( m_level_fine, id_fine ) );
407  Point p_coarse ( DWInterface<Problem::VAR, Problem::DIM>::get_position ( m_level_coarse, id_coarse ) );
408  Vector dist = ( p_fine.asVector() - p_coarse.asVector() ) / m_level_coarse->dCell();
409  double w[2] = { 1., 1. };
410  IntVector n[2] = { id_coarse, id_coarse };
411  const double & dx = dist[X];
412  if ( dx < 0. )
413  {
414  n[0][X] -= 1;
415  w[0] *= -dx;
416  w[1] *= 1 + dx;
417  }
418  else if ( dx > 0. )
419  {
420  n[1][X] += 1;
421  w[0] *= 1 - dx;
422  w[1] *= dx;
423  }
424  else
425  {
426  w[1] = 0.;
427  }
428  return w[0] * coarse_value ( n[0] ) +
429  w[1] * coarse_value ( n[1] );
430  }
