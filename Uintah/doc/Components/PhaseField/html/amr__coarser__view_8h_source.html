114  void
115  create_views (
116  DataWarehouse * dw,
117  const Level * level_fine,
118  const IntVector & low_fine,
119  const IntVector & high_fine
120  )
121  {
122  // Get coarser region under given fine region
123  const Level * level_coarse = level_fine->getCoarserLevel().get_rep();
124  IntVector l_coarse = AMRInterface<VAR, DIM>::get_coarser ( level_fine, low_fine );
125  IntVector h_coarse = AMRInterface<VAR, DIM>::get_coarser ( level_fine, high_fine - IntVector {1, 1, 1} ) + IntVector {1, 1, 1};
126 
127  // we need to include lower patches if l_coarse lies on a minus face
128  if ( VAR == NC )
129  l_coarse -= 1;
130 
131  // get list of coarse patches in region
132  std::vector<const Patch *> patches_coarse;
133  level_coarse->selectPatches ( l_coarse, h_coarse, patches_coarse );
134 
135  for ( const auto & patch_coarse : patches_coarse )
136  {
137  // virtual coarse patches
138  IntVector offset ( 0, 0, 0 );
139  if ( patch_coarse->isVirtual() )
140  offset = patch_coarse->getVirtualOffset();
141 
142  // need to check which BC applies on coarse level
143  Variable < PP, SubProblems<Problem> > subproblems_coarse;
144 
145  // get coarse subproblems (all logic for handling correctly all possible coarse geometries is already there)
146  dw->getOtherDataWarehouse ( Task::NewDW )->get ( subproblems_coarse, m_subproblems_label, m_material, patch_coarse );
147  auto problems_coarse = subproblems_coarse.get().get_rep();
148 
149  for ( const auto & p : *problems_coarse )
150  {
151  // check that the coarse problem is under fine region
152  IntVector low_coarse { Max ( p.get_low() + offset, l_coarse ) };
153  IntVector high_coarse { Min ( p.get_high() + offset, h_coarse ) };
154  if (
155  low_coarse[X] < high_coarse[X] &&
156  low_coarse[Y] < high_coarse[Y] &&
157  low_coarse[Z] < high_coarse[Z]
158  )
159  {
160  // get problem (virtual) view
161  if ( patch_coarse->isVirtual() )
162  this->m_views.push_back ( p.template get_view<I...> ().clone ( false, offset ) );
163  else
164  this->m_views.push_back ( p.template get_view<I...> ().clone ( false ) );
165 
166  // retrieve coarse data
167  this->m_views.back()->set ( dw, level_coarse, low_coarse, high_coarse );
168 
169  // add problem view support to view support
170  m_support.splice ( m_support.end(), this->m_views.back()->get_support() );
171  }
172  }
173  }
174 
175  // after all subviews are fetched simplify view support
176  m_support.simplify();
177  };
