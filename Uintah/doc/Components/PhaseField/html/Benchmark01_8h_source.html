579 template<VarType VAR, StnType STN>
580 void
581 Benchmark01<VAR, STN>::task_time_advance_postprocess (
582  const ProcessorGroup *,
583  const PatchSubset * patches,
584  const MaterialSubset *,
585  DataWarehouse * dw_old,
586  DataWarehouse * dw_new
587 )
588 {
589  dbg_out1 << "==== Benchmark01<VAR,STN>::task_time_advance_postprocess ====" << std::endl;
590 
591  for ( int p = 0; p < patches->size(); ++p )
592  {
593  const Patch * patch = patches->get ( p );
594  dbg_out2 << "== Patch: " << *patch << std::endl;
595 
596  DWFDView < ScalarField<const double>, STN, VAR > u_view ( dw_new, u_label, material, patch );
597 
598  IntVector i0;
599  if ( this->find_point ( patch, {M_PI, M_PI, 0.}, i0 ) )
600  {
601  double u0 = u_view[i0];
602  if ( fabs ( u0 ) > 2 ) SCI_THROW ( AssertionFailed ( "\n ERROR: Unstable simulation\n", __FILE__, __LINE__ ) );
603  dw_new->put ( sum_vartype ( u0 ), u0_label );
604  }
605  else
606  dw_new->put ( sum_vartype ( 0. ), u0_label );
607 
608  double energy = 0.;
609 
610  BlockRange range ( this->get_range ( patch ) );
611  dbg_out3 << "= Iterating over range " << range << std::endl;
612 
613  parallel_reduce_sum (
614  range,
615  [patch, &u_view, this] ( int i, int j, int k, double & energy )->void { time_advance_postprocess_energy ( {i, j, k}, patch, u_view, energy ); },
616  energy
617  );
618 
619  dw_new->put ( sum_vartype ( energy ), energy_label );
620  }
621 
622  dbg_out2 << std::endl;
623 }

627 template<VarType VAR, StnType STN>
628 void Benchmark01<VAR, STN>::initialize_solution (
629  const IntVector & id,
630  Patch const * patch,
631  View< ScalarField<double> > & u
632 )

635 
636  // BUG workaround
637  std::stringstream ss;
638  ss << v << std::endl;
639 
640  v[0] -= M_PI;

643  u[id] = tanh ( ( v.length() - 2. ) / ( epsilon * M_SQRT2 ) );
644 }

646 template<VarType VAR, StnType STN>
647 void Benchmark01<VAR, STN>::time_advance_solution (
648  const IntVector & id,
649  const FDView < ScalarField<const double>, STN > & u_old,
650  View< ScalarField<double> > & u_new
651 )
652 {
653  const double & u = u_old[id];
654  double lap = u_old.laplacian ( id );
655  double src = u * ( u * u - 1. );
656  double delta_u = delt * ( epsilon * epsilon * lap - src );
657  u_new[id] = u + delta_u;
658 }

660 template<VarType VAR, StnType STN>
661 void Benchmark01<VAR, STN>::time_advance_postprocess_energy (
662  const IntVector & id,
663  const Patch * patch,
664  const FDView < ScalarField<const double>, STN > & u_new,
665  double & energy
666 )
667 {
668  const double & u = u_new[id];
669  auto grad = u_new.gradient ( id );
670  double A = patch->getLevel()->dCell() [0] * patch->getLevel()->dCell() [1];
671  energy += A * ( epsilon * epsilon * ( grad[0] * grad[0] + grad[1] * grad[1] ) / 2. + ( u * u * u * u - 2 * u * u + 1. ) / 4. );
672 }
