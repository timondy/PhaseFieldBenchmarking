755 template<VarType VAR, StnType STN>
756 void Benchmark04<VAR, STN>::time_advance_v (
757  const IntVector & id,
758  const FDView < ScalarField<const double>, STN > & u_old,
759  View< ScalarField<double> > & v_new
760 )
761 {
762  const double & u = u_old[id];
763  double lap = u_old.laplacian ( id );
764  double src = u * ( u * u - 1. );
765  v_new[id] = - epsilon * epsilon * lap + src;
766 }

769 void Benchmark04<VAR, STN>::time_advance_u (

781 template<VarType VAR, StnType STN>
782 void Benchmark04<VAR, STN>::time_advance_postprocess_energy (
783  const IntVector & id,
784  const Patch * patch,
785  const FDView < ScalarField<const double>, STN > & u_new,
786  double & energy
787 )
788 {
789  const double & u = u_new[id];
790  auto grad = u_new.gradient ( id );
791  double A = patch->getLevel()->dCell() [0] * patch->getLevel()->dCell() [1];
792  energy += A * ( epsilon * epsilon * ( grad[0] * grad[0] + grad[1] * grad[1] ) / 2. + ( u * u * u * u - 2 * u * u + 1. ) / 4. );
793 }
