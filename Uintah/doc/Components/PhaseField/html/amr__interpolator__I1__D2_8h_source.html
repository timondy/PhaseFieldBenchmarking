136  Region
137  compute_fine_region (
138  const IntVector & low,
139  const IntVector & high
140  )
141  {
142  IntVector r = m_level_fine->getRefinementRatio();
143  IntVector l_ghosts {0, 0, 0}, h_ghosts {0, 0, 0};
144  for ( size_t d = 0; d < DIM; ++d )
145  {
146  int total = r[d] * ( E - 1 );
147  h_ghosts[d] = total / 2;
148  l_ghosts[d] = total - h_ghosts[d];
149  if ( VAR == NC )
150  {
151  if ( ! ( low[d] % r[d] ) && l_ghosts[d] > 0 ) l_ghosts[d] -= 1;
152  if ( ! ( ( high[d] - 1 ) % r[d] ) && h_ghosts[d] > 0 ) h_ghosts[d] -= 1;
153  }
154  }
155 
156  Region fine_region;
157  fine_region.low() = low - l_ghosts;
158  fine_region.high() = high + h_ghosts;
159 
160  return fine_region;
161  }

395  virtual V
396  operator[] (
397  const IntVector & id_fine
398  ) const override
399  {
400  IntVector id_coarse ( m_level_fine->mapCellToCoarser ( id_fine ) );
401  Point p_fine ( DWInterface<VAR, DIM>::get_position ( m_level_fine, id_fine ) );
402  Point p_coarse ( DWInterface<VAR, DIM>::get_position ( m_level_coarse, id_coarse ) );
403  Vector dist = ( p_fine.asVector() - p_coarse.asVector() ) / m_level_coarse->dCell();
404  double w[2][2] = {{ 1., 1. }, { 1., 1. }};
405  IntVector n[2][2] = {{ id_coarse, id_coarse }, { id_coarse, id_coarse }};
406  const double & dx = dist[X];
407  const double & dy = dist[Y];
408  if ( dx < 0. )
409  {
410  n[0][0][X] = n[0][1][X] -= 1;
411  w[0][0] *= -dx;
412  w[0][1] *= -dx;
413  w[1][0] *= 1 + dx;
414  w[1][1] *= 1 + dx;
415  }
416  else if ( dx > 0. )
417  {
418  n[1][0][X] = n[1][1][X] += 1;
419  w[0][0] *= 1 - dx;
420  w[0][1] *= 1 - dx;
421  w[1][0] *= dx;
422  w[1][1] *= dx;
423  }
424  else
425  {
426  w[1][0] = 0.;
427  w[1][1] = 0.;
428  }
429 
430  if ( dy < 0. )
431  {
432  n[0][0][Y] = n[1][0][Y] -= 1;
433  w[0][0] *= -dy;
434  w[1][0] *= -dy;
435  w[0][1] *= 1 + dy;
436  w[1][1] *= 1 + dy;
437  }
438  else if ( dy > 0. )
439  {
440  n[0][1][Y] = n[1][1][Y] += 1;
441  w[0][0] *= 1 - dy;
442  w[1][0] *= 1 - dy;
443  w[0][1] *= dy;
444  w[1][1] *= dy;
445  }
446  else
447  {
448  w[0][1] = 0.;
449  w[1][1] = 0.;
450  }
451 
452  return w[0][0] * coarse_value ( n[0][0] ) +
453  w[0][1] * coarse_value ( n[0][1] ) +
454  w[1][0] * coarse_value ( n[1][0] ) +
455  w[1][1] * coarse_value ( n[1][1] );
456  }
