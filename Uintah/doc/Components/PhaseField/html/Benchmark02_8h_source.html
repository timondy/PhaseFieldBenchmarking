734 
735  IntVector i2;

764 template<VarType VAR, StnType STN>
765 void Benchmark02<VAR, STN>::initialize_solution (
766  const IntVector & id,
767  Patch const * patch,
768  View< ScalarField<double> > & u
769 )
770 {
771  const int Ncirc = 7;
772  double xc[] = { 0.25, 0.125, 0.25, 0.5, 0.75, 0.5, 0.75 };
773  double yc[] = { 0.25, 0.375, 0.625, 0.125, 0.125, 0.5, 0.75 };
774  double rc[] = { 0.1, 0.0666666666667, 0.0666666666667, 0.05, 0.05, 0.125, 0.125 };
775 
776  Vector v ( this->get_position ( patch, id ).asVector() );
777 
778  // BUG workaround
779  std::stringstream ss;
780  ss << v << std::endl;
781 
782  for ( int i = 0; i < Ncirc; ++i )
783  {
784  Vector r ( v );
785  r[0] -= 2 * M_PI * xc[i];
786  r[1] -= 2 * M_PI * yc[i];
787  r[2] = 0.;
788  double dr = r.length() - 2 * M_PI * rc[i];
789  if ( dr < 0. )
790  {
791  u[id] = -1. + 2.*exp ( - ( epsilon * epsilon ) / ( dr * dr ) );
792  return;
793  }
794  }
795 
796  u[id] = -1.;
797 }

799 template<VarType VAR, StnType STN>
800 void Benchmark02<VAR, STN>::time_advance_v (
801  const IntVector & id,
802  const FDView < ScalarField<const double>, STN > & u_old,
803  View< ScalarField<double> > & v_new
804 )
805 {
806  const double & u = u_old[id];
807  double lap = u_old.laplacian ( id );
808  double src = u * ( u * u - 1. );
809  v_new[id] = - epsilon * epsilon * lap + src;
810 }

813 void Benchmark02<VAR, STN>::time_advance_u (
814  const IntVector & id,
815  const View< ScalarField<const double> > & u_old,

818 )

822  u_new[id] = u + delta_u;

825 template<VarType VAR, StnType STN>
826 void Benchmark02<VAR, STN>::time_advance_postprocess_energy (
827  const IntVector & id,
828  const Patch * patch,
829  const FDView < ScalarField<const double>, STN > & u_new,
830  double & energy
831 )
832 {
833  const double & u = u_new[id];
834  auto grad = u_new.gradient ( id );
835  double A = patch->getLevel()->dCell() [0] * patch->getLevel()->dCell() [1];
836  energy += A * ( epsilon * epsilon * ( grad[0] * grad[0] + grad[1] * grad[1] ) / 2. + ( u * u * u * u - 2 * u * u + 1. ) / 4. );
837 }
