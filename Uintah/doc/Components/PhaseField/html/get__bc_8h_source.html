73 template<typename T>
74 class get_bc< ScalarField<T> >
75 {
76  using Field = ScalarField<T>;
77 
78 public:
96  BCInfo<Field>
97  static exec (
98  Patch const * patch,
99  const Patch::FaceType & face,
100  const int & child,
101  const int & material,
102  const VarLabel * const & label,
103  const std::map<std::string, FC> * c2f,
104  bool & flag
105  )
106  {
107  BCInfo<Field> bci { T{}, BC::Unknown, FC::None };
108 
109  const std::string & desc = label->getName();
110  if ( patch->getBCType ( face ) == Patch::Coarse )
111  {
112  ASSERTMSG ( c2f, "null c2f map in get_bc" );
113  ASSERTMSG ( c2f->find ( desc ) != c2f->end(), "desc not found in c2f map in get_bc" );
114  bci.bc = BC::FineCoarseInterface;
115  bci.c2f = c2f->at ( desc );
116  }
117  else if ( patch->getBCType ( face ) == Patch::Neighbor )
118  {
119  bci.bc = BC::None;
120  }
121  else if ( const BCDataArray * data = patch->getBCDataArray ( face ) )
122  {
123  if ( const BoundCondBase * info = data->getBoundCondData ( material, desc, child ) )
124  {
125  if ( const BoundCond<double> * _info = dynamic_cast<const BoundCond<double> *> ( info ) )
126  {
127  bci.bc = str_to_bc ( _info->getBCType() );
128  bci.value = _info ->getValue();
129  }
130  delete info;
131  }
132  else if ( const BoundCondBase * info = data->getBoundCondData ( material, "Symmetric", child ) )
133  {
134  if ( info->getBCType() == "symmetry" )
135  bci.bc = BC::Symmetry;
136  delete info;
137  }
138  }
139 
140  switch ( bci.bc )
141  {
142  case BC::Unknown:
143  {
144  std::ostringstream msg;
145  msg << "\n ERROR: Unknown BC condition on patch " << *patch << " face " << face << std::endl;
146  SCI_THROW ( InvalidValue ( msg.str(), __FILE__, __LINE__ ) );
147  }
148  break;
149  case BC::None:
150  break;
151  default:
152  flag = true;
153  }
154  return bci;
155  }
156 };
