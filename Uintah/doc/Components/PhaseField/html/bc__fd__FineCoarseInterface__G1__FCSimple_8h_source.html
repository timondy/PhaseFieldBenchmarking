307  IntVector id1 { id };

309  if ( p0.asVector() [d] < p_fine.asVector() [d] )
310  id1[d] += 1;
311  else
312  id1[d] -= 1;
313  if ( m_fine_view->is_defined_at ( id1 ) )
314  {

323  u1 = coarse_value ( id1 );
324  }
