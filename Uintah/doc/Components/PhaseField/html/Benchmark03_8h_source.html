700 void
701 Benchmark03<VAR, STN>::task_time_advance_postprocess (
702  const ProcessorGroup *,
703  const PatchSubset * patches,
704  const MaterialSubset *,
705  DataWarehouse * dw_old,
706  DataWarehouse * dw_new
707 )
708 {
709  dbg_out1 << "==== Benchmark03<VAR,STN>::task_time_advance_postprocess ====" << std::endl;
710 
711  for ( int p = 0; p < patches->size(); ++p )
712  {
713  const Patch * patch = patches->get ( p );
714  dbg_out2 << "== Patch: " << *patch << std::endl;
715 
716  DWFDView < ScalarField<const double>, STN, VAR > u_view ( dw_new, u_label, material, patch );
717 
718  IntVector i0;
719  if ( this->find_point ( patch, {M_PI, 0., 0.}, i0 ) )
720  {
721  double u0 = u_view[i0];
722  if ( fabs ( u0 ) > 2 ) SCI_THROW ( AssertionFailed ( "\n ERROR: Unstable simulation\n", __FILE__, __LINE__ ) );
723  dw_new->put ( sum_vartype ( u0 ), u0_label );
724  }
725  else
726  dw_new->put ( sum_vartype ( 0. ), u0_label );
727 
728  double energy = 0.;
729 
730  BlockRange range ( this->get_range ( patch ) );
731  dbg_out3 << "= Iterating over range " << range << std::endl;
732 
733  parallel_reduce_sum (
734  range,
735  [patch, &u_view, this] ( int i, int j, int k, double & energy )->void { time_advance_postprocess_energy ( {i, j, k}, patch, u_view, energy ); },
736  energy
737  );
738 
739  dw_new->put ( sum_vartype ( energy ), energy_label );
740  }
741 
742  dbg_out2 << std::endl;

764 template<VarType VAR, StnType STN>
765 void Benchmark03<VAR, STN>::time_advance_v (
766  const IntVector & id,
767  const FDView < ScalarField<const double>, STN > & u_old,
768  View< ScalarField<double> > & v_new
769 )
770 {
771  const double & u = u_old[id];
772  double lap = u_old.laplacian ( id );
773  double src = u * ( u * u - 1. );
774  v_new[id] = - epsilon * epsilon * lap + src;
775 }

790 template<VarType VAR, StnType STN>
791 void Benchmark03<VAR, STN>::time_advance_postprocess_energy (
792  const IntVector & id,
793  const Patch * patch,
794  const FDView < ScalarField<const double>, STN > & u_new,
795  double & energy
796 )
797 {
798  const double & u = u_new[id];
799  auto grad = u_new.gradient ( id );
800  double A = patch->getLevel()->dCell() [0] * patch->getLevel()->dCell() [1];
801  energy += A * ( epsilon * epsilon * ( grad[0] * grad[0] + grad[1] * grad[1] ) / 2. + ( u * u * u * u - 2 * u * u + 1. ) / 4. );
802 }
