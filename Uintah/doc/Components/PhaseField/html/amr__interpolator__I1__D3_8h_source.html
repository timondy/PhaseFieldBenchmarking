139  Region
140  compute_fine_region (
141  const IntVector & low,
142  const IntVector & high
143  )
144  {
145  IntVector r = m_level_fine->getRefinementRatio();
146  IntVector l_ghosts {0, 0, 0}, h_ghosts {0, 0, 0};
147  for ( size_t d = 0; d < DIM; ++d )
148  {
149  int total = r[d] * ( E - 1 );
150  h_ghosts[d] = total / 2;
151  l_ghosts[d] = total - h_ghosts[d];
152  if ( VAR == NC )
153  {
154  if ( ! ( low[d] % r[d] ) && l_ghosts[d] > 0 ) l_ghosts[d] -= 1;
155  if ( ! ( ( high[d] - 1 ) % r[d] ) && h_ghosts[d] > 0 ) h_ghosts[d] -= 1;
156  }
157  }
158 
159  Region fine_region;
160  fine_region.low() = low - l_ghosts;
161  fine_region.high() = high + h_ghosts;
162 
163  return fine_region;
164  }
