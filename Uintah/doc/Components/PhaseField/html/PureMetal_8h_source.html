1189 )

1192  // them is scheduled, we need to prevent such task to be scheduled
1193  this->m_scheduler->overrideVariableBehavior ( subproblems_label->getName(), false, false, false, true, true );
1194

1200  pure_metal->require ( "delt", delt );
1201  pure_metal->require ( "alpha", alpha );
1202  pure_metal->require ( "R0", r0 );
1203  pure_metal->require ( "Delta", delta );
1204  pure_metal->require ( "epsilon", epsilon );

1206  pure_metal->getWithDefault ( "gamma_u", gamma_u, 1. );
1207

1210 
1211  if ( AMR )
1212  {
1213  this->setLockstepAMR ( true );
1214 
1215  // read amr parameters
1216  pure_metal->require ( "refine_threshold", refine_threshold );
1217

1219  if ( ! ( amr = params->findBlock ( "AMR" ) ) ) return;
1220  if ( ! ( fci = amr->findBlock ( "FineCoarseInterfaces" ) ) ) return;
1221  if ( ! ( fci = fci->findBlock ( "FCIType" ) ) ) return;

1225  fci->getAttribute ( "label", label );
1226  fci->getAttribute ( "var", var );
1227  c2f[label] = str_to_fc ( var );
1228  }
1229  while ( ( fci = fci->findNextBlock ( "FCIType" ) ) );
1230  }
1231 }
