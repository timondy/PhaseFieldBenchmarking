241  template <BCF Q>
242  bool adjust_dw_region (
243  IntVector & low,
244  IntVector & high
245  )
246  {
247  constexpr Patch::FaceType F = get_bcf<Q>::face;
248  constexpr DirType D = get_face<F>::dir;
249  constexpr int SGN = get_face<F>::sgn;
250  if ( SGN > 0 ) // F=D+
251  high[D] -= GN ;
252  else // F=D-
253  low[D] += GN;
254  return true;
255  }
